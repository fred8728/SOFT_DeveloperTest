import exceptions.CustomJsonException;
import org.json.JSONArray;
import org.json.JSONObject;

import java.util.ArrayList;
import java.util.List;

public class JSONParser {


    public List<String> parseToString(JSONObject obj, String key) {
        JSONArray arr = obj.getJSONArray(key);

        List<String> list = new ArrayList<String>();
        for (int i = 0; i < arr.length(); i++) {
            list.add(arr.getJSONObject(i).getString(key));
        }
        return list;
    }

    public String getSingleValueFromJson(JSONObject obj, String key) throws CustomJsonException {
        if (obj.isEmpty() || obj.isNull(key)){
            throw new CustomJsonException("JsonObject is null or empty");
        }
        return obj.getString(key);
    }

    public List<String> getAllValuesFromJson(JSONObject obj) {
        JSONArray arr = new JSONArray(obj);
        List <String> list = new ArrayList<>();
        for (int i = 0; i < arr.length(); i++) {
            list.add((String) arr.get(i));
        }
        return list;
    }
}
